model {
  # priors
  # lam and v > 0
  mu.b0 ~ dnorm(0, 1)
  tau.b0 <- 1/(sigma.b0 * sigma.b0)
  sigma.b0 ~ dgamma(0.5, 3)
  v ~ dgamma(1, 3)
  
  # Model
  for(i in 1:nind) { # individual
   for(k in 1:nobs[i])  {  # Occasion - Trap x day
	y[i,k] ~ dbern(p[i,k]) 
	cloglog(p[i,k]) <- log(lam[i,k]) + log(t[i,k]) + log(v) 
	lam[i,k] <- exp(b0[i])
	
	}
  }
  for(i in 1:nind) {
  b0[i] ~ dnorm(mu.b0, tau.b0)
  }
  
}
